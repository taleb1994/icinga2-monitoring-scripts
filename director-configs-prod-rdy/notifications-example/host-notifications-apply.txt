apply Notification "host-notification" to Host {
    command = "mail-host-notification"
    interval = 12h
    period = "24x7-timeperiod"
    assign where host.enable_notifications != "false"
    states = [ Down ]
    types = [
        Acknowledgement,
        Custom,
        DowntimeEnd,
        DowntimeRemoved,
        DowntimeStart,
        FlappingEnd,
        FlappingStart,
        Problem,
        Recovery
    ]
    users = [ "mohammed-taleb" ]
}
